struct Book {
    char title[MAX_TITLE_LEN];   // 书名
    char author[MAX_AUTHOR_LEN]; // 作者
    char publisher[MAX_PUBLISHER_LEN]; // 出版社
    char ISBN[MAX_ISBN_LEN];     // ISBN
    char genre[MAX_GENRE_LEN];   // 分类
    float price;                 // 价格
    int year;                    // 出版年份
    int quantity;                // 库存数量
};

struct User {
    char username[MAX_USERNAME_LEN]; // 用户名

    int isAdmin;                     // 是否为管理员

// 链表节点结构体
struct BookNode {
    Book book;                     // 图书信息
    struct BookNode* next;         // 指向下一个节点的指针
};

void addBook(BookManager* manager) {
    Book newBook;
    // 输入图书信息（验证逻辑）
    inputString("书名：", newBook.title, MAX_TITLE_LEN);
    // ISBN验证
    do {
        inputString("ISBN：", newBook.ISBN, MAX_ISBN_LEN);
    } while (!isValidISBN(newBook.ISBN) || 
             findBookByISBN(manager, newBook.ISBN) != NULL);
    
    // 创建节点并添加到链表头部
    BookNode* newNode = createBookNode(&newBook);
    newNode->next = manager->head;
    manager->head = newNode;
    manager->count++;
    printf("图书添加成功!\n");
}

// 从文件加载数据
void loadFromFile(BookManager *manager) {
    FILE *file = fopen(FILENAME, "rb");
    // 先销毁现有链表
    destroyManager(manager);
    
    int count;
    fread(&count, sizeof(int), 1, file);
    
    for (int i = 0; i < count; i++) {
        Book book;
        fread(&book, sizeof(Book), 1, file);
        
        BookNode *newNode = createBookNode(&book);
        newNode->next = manager->head;
        manager->head = newNode;
        manager->count++;
    }
    
    fclose(file);
    printf("数据加载成功！共加载 %d 本图书。\n", manager->count);
}
